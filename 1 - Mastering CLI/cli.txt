CLI - Command-Line Interface - Text-based

# Make directory
mkdir "directory name"

# Remove directory
rmdir "directory name"

# Change directory
cd "directory name"

# Make a file
touch "file name"

# Remove file
rm "file name"

# Present Working Directory
pwd

# List command 
ls

# Go back one directory
cd ..
